# Fraud Detection Signals for Stellar dApps

Fraud in Stellar dApps takes many forms: Sybil farming, wash trading, bot-driven referral abuse, and flash-loan manipulation. This guide documents the key signals to collect, how to score them, and what automated responses to trigger.

---

## Signal Categories

| Category            | Examples                                           |
| ------------------- | -------------------------------------------------- |
| **Velocity**        | Too many transactions in a short window            |
| **Account age**     | Brand-new account performing high-value actions    |
| **Graph anomalies** | Self-loop payments, circular payment graphs        |
| **Amount patterns** | Round-number amounts, exact duplicates             |
| **Behavioural**     | Machine-like timing (sub-second intervals)         |
| **Cross-context**   | Same IP, multiple wallets claiming the same reward |

## Signal 1 — Transaction Velocity

Count transactions per wallet per time window. High velocity with no legitimate use case is a red flag.

```ts
// lib/fraud.ts
import { Horizon } from '@stellar/stellar-sdk';

const server = new Horizon.Server(process.env.NEXT_PUBLIC_HORIZON_URL!);

export async function txVelocity(
  accountId: string,
  windowSeconds: number
): Promise<number> {
  const ops = await server
    .operations()
    .forAccount(accountId)
    .limit(200)
    .order('desc')
    .call();

  const cutoff = Date.now() - windowSeconds * 1000;
  return ops.records.filter((op) => new Date(op.created_at).getTime() > cutoff)
    .length;
}
```

## Signal 2 — Account Age

New accounts (low ledger sequence numbers) claiming rewards are suspicious:

```ts
export async function accountAgeLedgers(accountId: string): Promise<number> {
  const account = await server.loadAccount(accountId);
  // sequence encodes ledger at creation in the high 32 bits
  const creationLedger = Number(BigInt(account.sequenceNumber()) >> 32n);
  const currentLedger = (await server.ledgers().order('desc').limit(1).call())
    .records[0].sequence;
  return currentLedger - creationLedger;
}

// Flag accounts created < 1000 ledgers ago (~83 minutes)
export async function isNewAccount(accountId: string): Promise<boolean> {
  return (await accountAgeLedgers(accountId)) < 1000;
}
```

## Signal 3 — Circular Payment Detection

A → B → A (or longer cycles) is a wash-trading pattern. Track payment destinations from recent operations:

```ts
export async function detectCircularPayments(
  accountId: string,
  depth = 2
): Promise<boolean> {
  const payments = await server
    .payments()
    .forAccount(accountId)
    .limit(20)
    .order('desc')
    .call();

  const destinations = new Set<string>();
  for (const p of payments.records) {
    if (p.type !== 'payment') continue;
    const dest = (p as any).to as string;
    if (dest === accountId) return true; // direct self-payment
    if (depth > 1) {
      // Check if the destination recently paid back to us
      const reverse = await server
        .payments()
        .forAccount(dest)
        .limit(10)
        .order('desc')
        .call();
      const reversed = reverse.records.some(
        (r) => r.type === 'payment' && (r as any).to === accountId
      );
      if (reversed) return true;
    }
    destinations.add(dest);
  }
  return false;
}
```

## Signal 4 — Round Amount Pattern

Exact round-number amounts (100, 1000, 10000) in a series of payments can indicate bot activity:

```ts
export function isRoundAmount(amount: string): boolean {
  const n = parseFloat(amount);
  return n > 0 && n % 10 === 0;
}

export async function roundAmountRatio(accountId: string): Promise<number> {
  const payments = await server
    .payments()
    .forAccount(accountId)
    .limit(50)
    .order('desc')
    .call();

  const paymentOps = payments.records.filter((p) => p.type === 'payment');
  if (paymentOps.length === 0) return 0;

  const roundCount = paymentOps.filter((p) =>
    isRoundAmount((p as any).amount)
  ).length;
  return roundCount / paymentOps.length;
}
```

## Signal 5 — Timing Regularity (Bot Detection)

Humans don't transact at exactly 1-second intervals. Measure the coefficient of variation of time gaps:

```ts
export async function timingRegularity(accountId: string): Promise<number> {
  const ops = await server
    .operations()
    .forAccount(accountId)
    .limit(20)
    .order('desc')
    .call();

  const times = ops.records.map((o) => new Date(o.created_at).getTime());
  if (times.length < 3) return 0;

  const gaps = times.slice(0, -1).map((t, i) => t - times[i + 1]);
  const mean = gaps.reduce((a, b) => a + b, 0) / gaps.length;
  const variance = gaps.reduce((a, b) => a + (b - mean) ** 2, 0) / gaps.length;
  const cv = Math.sqrt(variance) / mean; // coefficient of variation
  // Low CV (< 0.1) = very regular timing = likely bot
  return cv;
}
```

## Composite Fraud Score

Combine signals into a 0–100 score:

```ts
export async function fraudScore(accountId: string): Promise<number> {
  const [velocity, isNew, circular, roundRatio, timing] = await Promise.all([
    txVelocity(accountId, 300), // last 5 minutes
    isNewAccount(accountId),
    detectCircularPayments(accountId),
    roundAmountRatio(accountId),
    timingRegularity(accountId),
  ]);

  let score = 0;
  if (velocity > 20) score += 30;
  if (isNew) score += 25;
  if (circular) score += 30;
  if (roundRatio > 0.8) score += 10;
  if (timing < 0.1) score += 20;

  return Math.min(score, 100);
}
```

## Automated Response Actions

| Score  | Action                                                       |
| ------ | ------------------------------------------------------------ |
| 0–29   | Allow — normal                                               |
| 30–59  | Flag for review; add CAPTCHA on next action                  |
| 60–79  | Rate-limit to 1 action/minute; notify ops team               |
| 80–100 | Block account from claiming rewards; queue for manual review |

```ts
export async function enforcePolicy(accountId: string) {
  const score = await fraudScore(accountId);

  if (score >= 80) {
    await blockAccount(accountId, 'fraud_score_critical');
    return { allowed: false, reason: 'Account flagged for review' };
  }
  if (score >= 60) {
    await rateLimitAccount(accountId, 60_000); // 1/min
    return { allowed: true, rateLimit: true };
  }
  return { allowed: true };
}
```
